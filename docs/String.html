def affix? prefix, suffix=nil
  suffix = prefix if suffix.nil?
  if Regexp === prefix
    md = match(/\A#{prefix}/)
    if md
      rest = md.post_match
    else
      return false
    end
  else
    prefix = prefix.to_s
    if start_with? prefix
      i = prefix.length
      rest = self[i..-1]
    else
      return false
    end
  end
  rest.suffix? suffix
end

def suffix? pattern
  if Regexp === pattern
    #match?(/#{pattern}\z/)
